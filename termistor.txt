//NTC110 100kQ 5%
int lämpötila = 0; 
int Vo; 
float R1 = 10000; 
float logR2, R2, T; 
float cl = 1.009249522e-03, c2 = 2.378405444e-04, c3 = 2.019202697e-07; 
void setup() { Serial.begin(9600); }
void loop() { 
 Vo = analogRead(lämpötila); 
 R2 = R1 * (1023.0 / (float)Vo - 1.0); 
 logR2 = log(R2); 
 T = (1.0 / (cl + c2*logR2 + c3*logR2*logR2*logR2)); 
 T = T - 273.15; 
 Serial.print("lämpötila: "); Serial.print(T); Serial.println(" C"); 
 delay(500); 
}
